 124     /**                                                                                                  
 125      * Submit a new transaction to the ledger.                                                           
 126      *                                                                                                   
 127      * @param bearerToken  HTTP request 'Authorization' header                                           
 128      * @param transaction  transaction to submit                                                         
 129      *                                                                                                   
 130      * @return  HTTP Status 200 if transaction was successfully submitted                                
 131      */                                                                                                  
 132     @PostMapping(value = "/transactions", consumes = "application/json")                                 
 133     @ResponseStatus(HttpStatus.OK)                                                                       
 134     public ResponseEntity<?> addTransaction(                                                             
 135             @RequestHeader("Authorization") String bearerToken,                                          
 136             @RequestBody Transaction transaction) {                                                      
 137         if (bearerToken != null && bearerToken.startsWith("Bearer ")) {                                  
 138             bearerToken = bearerToken.split("Bearer ")[1];                                               
 139         }                                                                                                
 140         try {                                                                                            
 141             if (bearerToken == null) {                                                                   
 142                 LOGGER.error("Transaction submission failed: "                                           
 143                     + "Authorization header null");                                                      
 144                 throw new IllegalArgumentException(                                                      
 145                         EXCEPTION_MESSAGE_WHEN_AUTHORIZATION_HEADER_NULL);                               
 146             }                                                                                            
 147             final DecodedJWT jwt = this.verifier.verify(bearerToken);                                    
 148                                                                                                          
 149             // Check against cache for duplicate transactions                                            
 150             if (this.cache.asMap().containsKey(transaction.getRequestUuid())) {                          
 151                 throw new IllegalStateException(                                                         
 152                         EXCEPTION_MESSAGE_DUPLICATE_TRANSACTION);                                        
 153             }                                                                                            
 154                                                                                                          
 155             // validate transaction                                                                      
 156             transactionValidator.validateTransaction(localRoutingNum,                                    
 157                     jwt.getClaim(JWT_ACCOUNT_KEY).asString(), transaction);                              
 158             // Ensure sender balance can cover transaction.                                              
 159             if (transaction.getFromRoutingNum().equals(localRoutingNum)) {                               
 160                 int balance = getAvailableBalance(                                                       
 161                         bearerToken, transaction.getFromAccountNum());                                   
 162                 if (balance < transaction.getAmount()) {                                                 
 163                     LOGGER.error("Transaction submission failed: "                                       
 164                         + "Insufficient balance");                                                       
 165                     throw new IllegalStateException(                                                     
 166                             EXCEPTION_MESSAGE_INSUFFICIENT_BALANCE);                                     
 167                 }                                                                                        
 168             }                                                                                            
 169                                                                                                          
 170             // No exceptions thrown. Add to ledger                                                       
 171             transactionRepository.save(transaction);                                                     
 172             this.cache.put(transaction.getRequestUuid(),                                                 
 173                     transaction.getTransactionId());                                                     
 174             LOGGER.info("Submitted transaction successfully");                                           
 175             return new ResponseEntity<String>(READINESS_CODE,                                            
 176                     HttpStatus.CREATED);                                                                 
 177                                                                                                          
 178         } catch (JWTVerificationException e) {                                                           
 179             LOGGER.error("Failed to submit transaction: "                                                
 180                 + "not authorized");                                                                     
 181             return new ResponseEntity<String>(UNAUTHORIZED_CODE,                                         
 182                                               HttpStatus.UNAUTHORIZED);                                  
 183         } catch (IllegalArgumentException | IllegalStateException e) {                                   
 184             LOGGER.error("Failed to retrieve account balance: "                                          
 185                 + "bad request");                                                                        
 186             return new ResponseEntity<String>(e.getMessage(),                                            
 187                                               HttpStatus.BAD_REQUEST);                                   
 188         } catch (ResourceAccessException                                                                 
 189                 | CannotCreateTransactionException                                                       
 190                 | HttpServerErrorException e) {                                                          
 191             LOGGER.error("Failed to retrieve account balance");                                          
 192             return new ResponseEntity<String>(e.getMessage(),                                            
 193                                               HttpStatus.INTERNAL_SERVER_ERROR);                         
 194         }                                                                                                
 195     }
