 309      private Set<String> keys(JedisCluster jedisCluster, String pattern){                                          
























 310          Set<String> keys = new TreeSet<>();                                                                       
 311          Map<String, JedisPool> clusterNodes = jedisCluster.getClusterNodes();                                     

 312          for(String k : clusterNodes.keySet()){                                                                    
 313              JedisPool jp = clusterNodes.get(k);                                                                   
 314              Jedis connection = jp.getResource();                                                                  
 315              try {                                                                                                 
 316                  keys.addAll(connection.keys(pattern));                                                            

 317              } catch (Exception e){                                                                                
 318                  LOG.error("Getting keys error: {}", e);                                                           
 319              } finally {                                                                                           
 320                  connection.close();                                                                               
 321              }                                                                                                     
 322          }                                                                                                         
 323          return keys;                                                                                              
 324      }
