 398      private boolean cancelInner(@Nullable JobRequest request) {                                                   
 399          if (request != null) {                                                                                    
 400              CAT.i("Found pending job %s, canceling", request);                                                    
 401              getJobProxy(request).cancel(request.getJobId());                                                      
 402              getJobStorage().remove(request);                                                                      
 403              request.setScheduledAt(0); // reset value                                                             
 404              return true;                                                                                          
 405          } else {                                                                                                  
 406              return false;                                                                                         
 407          }                                                                                                         
 408      }

 410      private boolean cancelInner(@Nullable Job job) {                                                              
 411          if (job != null && !job.isFinished() && !job.isCanceled()) {                                              
 412              CAT.i("Cancel running %s", job);                                                                      
 413              job.cancel(true);                                                                                     
 414              return true;                                                                                          
 415          } else {                                                                                                  
 416              return false;                                                                                         
 417          }                                                                                                         
 418      }

 421          int canceled = 0;

 423          Set<JobRequest> requests = mJobStorage.getAllJobRequests(tag, true);                                      

 424          for (JobRequest request : requests) {                                                                     
 425              if (cancelInner(request)) {

 431          Set<Job> jobs = TextUtils.isEmpty(tag) ? getAllJobs() : getAllJobsForTag(tag);

 434                  canceled++;                                                                                       
 435              }                                                                                                     
 436          }                                                                                                         
 437          return canceled;

 591      private static void sendAddJobCreatorIntent(@NonNull Context context) {                                       
 592          Intent intent = new Intent(JobCreator.ACTION_ADD_JOB_CREATOR);                                            
 593          List<ResolveInfo> resolveInfos = context.getPackageManager().queryBroadcastReceivers(intent, 0);          
 594          String myPackage = context.getPackageName();                                                              
 595                                                                                                                    
 596          for (ResolveInfo resolveInfo : resolveInfos) {                                                            
 597              ActivityInfo activityInfo = resolveInfo.activityInfo;                                                 
 598              if (activityInfo == null || activityInfo.exported || !myPackage.equals(activityInfo.packageName)      
 599                      || TextUtils.isEmpty(activityInfo.name)) {                                                    
 600                  continue;                                                                                         
 601              }                                                                                                     
 602                                                                                                                    
 603              try {                                                                                                 
 604                  JobCreator.AddJobCreatorReceiver receiver =                                                       
 605                          (JobCreator.AddJobCreatorReceiver) Class.forName(activityInfo.name).newInstance();        
 606                                                                                                                    
 607                  receiver.addJobCreator(context, instance);                                                        
 608              } catch (Exception ignored) {                                                                         
 609              }                                                                                                     
 610          }                                                                                                         
 611      }
