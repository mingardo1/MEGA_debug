 438      private int extractIntPart(String str) {                                                                      
 439          int result = 0, length = str.length();                                                                    
 440          for (int offset = 0; offset < length; offset++) {                                                         
 441              char c = str.charAt(offset);                                                                          
 442              if ('0' <= c && c <= '9')                                                                             
 443                  result = result * 10 + (c - '0');                                                                 
 444              else                                                                                                  
 445                  break;                                                                                            
 446          }                                                                                                         
 447          return result;                                                                                            
 448      }
