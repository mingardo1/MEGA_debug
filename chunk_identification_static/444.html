 128     private static String[] parseJson(String[] args) {                                                   
 129         BufferedReader reader = null;                                                                    
 130         String lastStr = "";                                                                             
 131         try {                                                                                            
 132             FileInputStream fileInputStream = new FileInputStream(args[0]);                              
 133             InputStreamReader inputStreamReader = new InputStreamReader(fileInputStream, "UTF-8");       
 134             reader = new BufferedReader(inputStreamReader);                                              
 135             String tempString = null;                                                                    
 136             while ((tempString = reader.readLine()) != null) {                                           
 137                 lastStr += tempString;                                                                   
 138             }                                                                                            
 139             reader.close();                                                                              
 140         } catch (IOException e) {                                                                        
 141             e.printStackTrace();                                                                         
 142         } finally {                                                                                      
 143             if (reader != null) {                                                                        
 144                 try {                                                                                    
 145                     reader.close();                                                                      
 146                 } catch (IOException e) {                                                                
 147                     e.printStackTrace();                                                                 
 148                 }                                                                                        
 149             }                                                                                            
 150         }                                                                                                
 151         Map<String, Object> map = JSON.parseObject(lastStr, new TypeReference<Map<String, Object>>() {   
 152         });                                                                                              
 153         List<String> list = new LinkedList<>();                                                          
 154                                                                                                          
 155         for (Map.Entry<String, Object> entry : map.entrySet()) {                                         
 156             list.add("-" + entry.getKey());                                                              
 157             list.add(entry.getValue().toString());                                                       
 158         }                                                                                                
 159         String[] array = list.toArray(new String[list.size()]);                                          
 160         return array;
