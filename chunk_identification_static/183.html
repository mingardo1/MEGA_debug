 297             private void updatePanelOffset(int progress) {

 300                 mSearchOverlay.setTranslationX(- panelWidth + offset);
 301             }

 305                 if (mProgress > 25 && mLauncherOverlayCallbacks.enterFullImmersion()) {
 306                     ObjectAnimator oa = LauncherAnimUtils.ofFloat(mSearchOverlay, "translationX", 0);
 307                     oa.addListener(new AnimatorListenerAdapter() {
 308                         @Override

 310                             mSearchOverlay.setLayerType(View.LAYER_TYPE_NONE, null);
