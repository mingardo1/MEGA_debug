  30      public static void setTheme(Activity activity) {                                                              
  31              // if we have a data uri that sets the theme let's do it here                                         
  32          Uri data = activity.getIntent().getData();                                                                
  33          if (data != null) {                                                                                       
  34              if (data.getAuthority().equals(PREFERENCES_URI_AUTHORITY)) {                                          
  35                  List<String> segments = data.getPathSegments();                                                   
  36                                                                                                                    
  37                  // check if we have reached /preferences/theme                                                    
  38                  if (segments.size() > 0 && segments.get(0).equals(URI_SEGMENT_THEME)) {                           
  39                                                                                                                    
  40                      // activate the theme preference                                                              
  41                      SharedPreferences preferences = PreferenceManager.getDefaultSharedPreferences(activity);      
  42                      SharedPreferences.Editor editor = preferences.edit();                                         
  43                      editor.putBoolean(PrefUtils.PREF_THEME_MODIFIED, true);                                       
  44                                                                                                                    
  45                      editor.apply();                                                                               
  46                  }                                                                                                 
  47              }                                                                                                     
  48          }                                                                                                         
  49                                                                                                                    
  50          switch (PrefUtils.getIntPref(activity, PrefUtils.PREF_THEME, THEME_SYSTEM)) {                             
  51              case THEME_AUTO:                                                                                      
  52                  AppCompatDelegate.setDefaultNightMode(AppCompatDelegate.MODE_NIGHT_AUTO);                         
  53                  break;                                                                                            
  54              case THEME_DARK:                                                                                      
  55                  AppCompatDelegate.setDefaultNightMode(AppCompatDelegate.MODE_NIGHT_YES);                          
  56                  break;                                                                                            
  57              case THEME_LIGHT:                                                                                     
  58                  AppCompatDelegate.setDefaultNightMode(AppCompatDelegate.MODE_NIGHT_NO);                           
  59                  break;                                                                                            
  60              case THEME_SYSTEM:                                                                                    
  61                  AppCompatDelegate.setDefaultNightMode(AppCompatDelegate.MODE_NIGHT_FOLLOW_SYSTEM);                
  62                  break;                                                                                            
  63          }

  79      public static int getOptimalDrawerWidth(Context context) {                                                    
  80          Point displaySize = DisplayUtils.getDisplayPixelSize(context);                                            
  81          int appBarHeight = DisplayUtils.getActionBarHeight(context);                                              
  82          int drawerWidth = Math.min(displaySize.x, displaySize.y) - appBarHeight;                                  
  83          int maxDp = (DisplayUtils.isXLarge(context) ? 400 : 320);                                                 
  84          int maxPx = DisplayUtils.dpToPx(context, maxDp);                                                          
  85          return Math.min(drawerWidth, maxPx);                                                                      
  86      }
