 299             List<ResolveInfo> resolveInfos = context.getPackageManager().queryIntentServices(intent, 0); 
 300             return resolveInfos != null && !resolveInfos.isEmpty();                                      
 301         } catch (Exception e) {                                                                          
 302             return false;
