  86     public void parseSelectFields(JoinInfo joinInfo){                                                    
  87         String sideTableName = joinInfo.getSideTableName();                                              
  88         String nonSideTableName = joinInfo.getNonSideTable();                                            
  89         List<String> fields = Lists.newArrayList();                                                      
  90         int sideTableFieldIndex = 0;                                                                     
  91                                                                                                          
  92         for( int i=0; i<outFieldInfoList.size(); i++){                                                   
  93             FieldInfo fieldInfo = outFieldInfoList.get(i);                                               
  94             if(fieldInfo.getTable().equalsIgnoreCase(sideTableName)){                                    
  95                 String sideFieldName = sideTableInfo.getPhysicalFields().getOrDefault(fieldInfo.getFieldNðŸ”µ
  96                 fields.add(sideFieldName);                                                               
  97                 sideSelectFieldsType.put(sideTableFieldIndex, getTargetFieldType(sideFieldName));        
  98                 sideFieldIndex.put(i, sideTableFieldIndex);                                              
  99                 sideFieldNameIndex.put(i, sideFieldName);                                                
 100                 sideTableFieldIndex++;                                                                   
 101             }else if(fieldInfo.getTable().equalsIgnoreCase(nonSideTableName)){                           
 102                 int nonSideIndex = rowTypeInfo.getFieldIndex(fieldInfo.getFieldName());                  
 103                 inFieldIndex.put(i, nonSideIndex);                                                       
 104             }else{                                                                                       
 105                 throw new RuntimeException("unknown table " + fieldInfo.getTable());                     
 106             }                                                                                            
 107         }                                                                                                
 108                                                                                                          
 109         if(fields.size() == 0){                                                                          
 110             throw new RuntimeException("select non field from table " +  sideTableName);                 
 111         }                                                                                                
 112                                                                                                          
 113         sideSelectFields = String.join(",", fields);                                                     
 114     }

 126     public void dealOneEqualCon(SqlNode sqlNode, String sideTableName){                                  
 127         if(!SqlKind.COMPARISON.contains(sqlNode.getKind())){                                             
 128             throw new RuntimeException("not compare operator.");                                         
 129         }                                                                                                
 130                                                                                                          
 131         SqlIdentifier left = (SqlIdentifier)((SqlBasicCall)sqlNode).getOperands()[0];                    
 132         SqlIdentifier right = (SqlIdentifier)((SqlBasicCall)sqlNode).getOperands()[1];                   
 133                                                                                                          
 134         String leftTableName = left.getComponent(0).getSimple();                                         
 135         String leftField = left.getComponent(1).getSimple();                                             
 136                                                                                                          
 137         String rightTableName = right.getComponent(0).getSimple();                                       
 138         String rightField = right.getComponent(1).getSimple();                                           
 139                                                                                                          
 140         if(leftTableName.equalsIgnoreCase(sideTableName)){                                               
 141             equalFieldList.add(leftField);                                                               
 142             int equalFieldIndex = -1;                                                                    
 143             for(int i=0; i<rowTypeInfo.getFieldNames().length; i++){                                     
 144                 String fieldName = rowTypeInfo.getFieldNames()[i];                                       
 145                 if(fieldName.equalsIgnoreCase(rightField)){                                              
 146                     equalFieldIndex = i;                                                                 
 147                 }                                                                                        
 148             }                                                                                            
 149             if(equalFieldIndex == -1){                                                                   
 150                 throw new RuntimeException("can't find equal field " + rightField);                      
 151             }                                                                                            
 152                                                                                                          
 153             equalValIndex.add(equalFieldIndex);                                                          
 154                                                                                                          
 155         }else if(rightTableName.equalsIgnoreCase(sideTableName)){                                        
 156                                                                                                          
 157             equalFieldList.add(rightField);                                                              
 158             int equalFieldIndex = -1;                                                                    
 159             for(int i=0; i<rowTypeInfo.getFieldNames().length; i++){                                     
 160                 String fieldName = rowTypeInfo.getFieldNames()[i];                                       
 161                 if(fieldName.equalsIgnoreCase(leftField)){                                               
 162                     equalFieldIndex = i;                                                                 
 163                 }                                                                                        
 164             }                                                                                            
 165             if(equalFieldIndex == -1){                                                                   
 166                 throw new RuntimeException("can't find equal field " + rightField);                      
 167             }                                                                                            
 168                                                                                                          
 169             equalValIndex.add(equalFieldIndex);                                                          
 170                                                                                                          
 171         }else{                                                                                           
 172             throw new RuntimeException("resolve equalFieldList error:" + sqlNode.toString());            
 173         }                                                                                                
 174     }
