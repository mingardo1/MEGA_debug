  40  public abstract class AbstractTableInfo implements Serializable {
  41  
  42      public static final String PARALLELISM_KEY = "parallelism";
  43      public static final String ERROR_LIMIT = "errorLimit";
  44      private final List<String> fieldList = Lists.newArrayList();
  45      private final List<String> fieldTypeList = Lists.newArrayList();
  46      private final List<Class> fieldClassList = Lists.newArrayList();
  47      private final List<FieldExtraInfo> fieldExtraInfoList = Lists.newArrayList();
  48      private String name;
  49      private String type;
  50      private String[] fields;
  51      private String[] fieldTypes;
  52      private Class<?>[] fieldClasses;
  53      /**
  54       * key:别名, value: realField
  55       */
  56      private Map<String, String> physicalFields = Maps.newLinkedHashMap();
  57      private List<String> primaryKeys;
  58  
  59      private Integer parallelism = -1;
  60      private Map<String, String> checkProperties;
  61  
  62      /**
  63       * 构建脏数据插件的相关信息
  64       */
  65      private Map<String, Object> dirtyProperties;
  66  
  67      /**
  68       * error data limit. Task will failed once {@link AbstractDtRichOutputFormat#outDirtyRecords}
  69       * count over limit. Default 1000L;
  70       */
  71 -    private Long errorLimit = 0L;
  72 +    private Long errorLimit = Long.MAX_VALUE;

  74      public String[] getFieldTypes() {
  75          return fieldTypes;
  76      }
  77  
  78      public void setFieldTypes(String[] fieldTypes) {
  79          this.fieldTypes = fieldTypes;
  80      }
  81  
  82      public abstract boolean check();
  83  
  84      public String getType() {
  85          return type;
  86      }
  87  
  88      public void setType(String type) {
  89          this.type = type;
  90      }
  91  
  92      public String[] getFields() {
  93          return fields;
  94      }
  95  
  96      public void setFields(String[] fields) {
  97          this.fields = fields;
  98      }
  99  
 100      public Class<?>[] getFieldClasses() {
 101          return fieldClasses;
 102      }
 103  
 104      public void setFieldClasses(Class<?>[] fieldClasses) {
 105          this.fieldClasses = fieldClasses;
 106      }
 107  
 108      public List<String> getPrimaryKeys() {
 109          if (this instanceof AbstractSideTableInfo &&
 110              Objects.isNull(primaryKeys)) {
 111              throw new IllegalArgumentException("Side table must contain [primary key]!");
 112          }
 113          return primaryKeys;
 114      }
 115  
 116      public void setPrimaryKeys(List<String> primaryKeys) {
 117          this.primaryKeys = primaryKeys;
 118      }
 119  
 120      public String getName() {
 121          return name;
 122      }
 123  
 124      public void setName(String name) {
 125          this.name = name;
 126      }
 127  
 128      public Integer getParallelism() {
 129          return parallelism;
 130      }
 131  
 132      public void setParallelism(Integer parallelism) {
 133          if (parallelism == null) {
 134              return;
 135          }
 136  
 137          if (parallelism <= 0) {
 138              throw new RuntimeException("Abnormal parameter settings: parallelism > 0");
 139          }
 140  
 141          this.parallelism = parallelism;
 142      }
 143  
 144      public void addField(String fieldName) {
 145          if (fieldList.contains(fieldName)) {
 146              throw new RuntimeException("redundancy field name " + fieldName + " in table " + getName());
 147          }
 148  
 149          fieldList.add(fieldName);
 150      }
 151  
 152      public void addPhysicalMappings(String aliasName, String physicalFieldName) {
 153          physicalFields.put(aliasName, physicalFieldName);
 154      }
 155  
 156      public void addFieldClass(Class fieldClass) {
 157          fieldClassList.add(fieldClass);
 158      }
 159  
 160      public void addFieldType(String fieldType) {
 161          fieldTypeList.add(fieldType);
 162      }
 163  
 164      public void addFieldExtraInfo(FieldExtraInfo extraInfo) {
 165          fieldExtraInfoList.add(extraInfo);
 166      }
 167  
 168      public List<String> getFieldList() {
 169          return fieldList;
 170      }
 171  
 172      public List<String> getFieldTypeList() {
 173          return fieldTypeList;
 174      }
 175  
 176      public List<Class> getFieldClassList() {
 177          return fieldClassList;
 178      }
 179  
 180      public Map<String, String> getPhysicalFields() {
 181          return physicalFields;
 182      }
 183  
 184      public void setPhysicalFields(Map<String, String> physicalFields) {
 185          this.physicalFields = physicalFields;
 186      }
 187  
 188      public List<FieldExtraInfo> getFieldExtraInfoList() {
 189          return fieldExtraInfoList;
 190      }
 191  
 192      public Map<String, String> getCheckProperties() {
 193          return checkProperties;
 194      }
 195  
 196      public void setCheckProperties() {
 197          this.checkProperties = buildCheckProperties();
 198      }
 199  
 200      public Map<String, String> buildCheckProperties() {
 201          return Maps.newHashMap();
 202      }
 203  
 204      public Map<String, Object> getDirtyProperties() {
 205          dirtyProperties.put(DirtyKeys.TABLE_NAME, this.name);
 206          return dirtyProperties;
 207      }
 208  
 209      public void setDirtyProperties(Map<String, Object> dirtyProperties) {
 210          this.dirtyProperties = dirtyProperties;
 211      }
 212  
 213      public Long getErrorLimit() {
 214          return errorLimit;
 215      }
 216  
 217      public void setErrorLimit(Long errorLimit) {
 218          this.errorLimit = errorLimit;
 219      }
 220  
 221      public void finish() {
 222          this.fields = fieldList.toArray(new String[0]);
 223          this.fieldClasses = fieldClassList.toArray(new Class[0]);
 224          this.fieldTypes = fieldTypeList.toArray(new String[0]);
 225      }
 226  
 227      /**
 228       * field extra info，used to store `not null` `default 0`...，
 229       * <p>
 230       * now, only support not null
 231       */
 232      public static class FieldExtraInfo implements Serializable {
 233  
 234          /**
 235           * default false：allow field is null
 236           */
 237          boolean notNull = false;
 238          /**
 239           * field length,eg.char(4)
 240           */
 241          int length;
 242  
 243          public int getLength() {
 244              return length;
 245          }
 246  
 247          public void setLength(int length) {
 248              this.length = length;
 249          }
 250  
 251          public boolean getNotNull() {
 252              return notNull;
 253          }
 254  
 255          public void setNotNull(boolean notNull) {
 256              this.notNull = notNull;
 257          }
 258  
 259          @Override
 260          public String toString() {
 261              return "FieldExtraInfo{" +
 262                      "notNull=" + notNull +
 263                      ", length=" + length +
 264                      '}';
 265          }
 266      }
