 208     static byte[] flattenBitmap(Bitmap bitmap) {                                                         
 209         // Try go guesstimate how much space the icon will take when serialized                          
 210         // to avoid unnecessary allocations/copies during the write.                                     
 211         int size = bitmap.getWidth() * bitmap.getHeight() * 4;                                           
 212         ByteArrayOutputStream out = new ByteArrayOutputStream(size);                                     
 213         try {                                                                                            
 214             bitmap.compress(Bitmap.CompressFormat.PNG, 100, out);                                        
 215             out.flush();                                                                                 
 216             out.close();                                                                                 
 217             return out.toByteArray();                                                                    
 218         } catch (IOException e) {                                                                        
 219             Log.w("Favorite", "Could not write icon");                                                   
 220             return null;                                                                                 
 221         }
