 194      static byte[] flattenBitmap(Bitmap bitmap) {
 195          // Try go guesstimate how much space the icon will take when serialized
 196          // to avoid unnecessary allocations/copies during the write.
 197          int size = bitmap.getWidth() * bitmap.getHeight() * 4;
 198          ByteArrayOutputStream out = new ByteArrayOutputStream(size);
 199          try {
 200              bitmap.compress(Bitmap.CompressFormat.PNG, 100, out);
 201              out.flush();
 202              out.close();
 203              return out.toByteArray();
 204          } catch (IOException e) {
 205              Log.w("Favorite", "Could not write icon");
 206              return null;
 207          }
 208      }
