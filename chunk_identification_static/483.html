 199                      }
 200                  }
 201              }
 202              queryAndFillData(tmpCache, connection);
 203          } catch (Exception e) {
 204              LOG.error("", e);
 205              throw new SQLException(e);
 206          } finally {
 207              if (connection != null) {
 208                  connection.close();
 209              }
 210          }
 211      }

 213      private void queryAndFillData(Map<String, List<Map<String, Object>>> tmpCache, Connection connection) throws S🔵
 214          //load data from table
 215          String sql = sideInfo.getSqlCondition();
 216          Statement statement = connection.createStatement();
 217          statement.setFetchSize(getFetchSize());
 218          ResultSet resultSet = statement.executeQuery(sql);
 219  
 220          String[] sideFieldNames = StringUtils.split(sideInfo.getSideSelectFields(), ",");
 221          String[] fields = sideInfo.getSideTableInfo().getFieldTypes();
 222          while (resultSet.next()) {
 223              Map<String, Object> oneRow = Maps.newHashMap();
 224              for (String fieldName : sideFieldNames) {
 225                  Object object = resultSet.getObject(fieldName.trim());
 226                  int fieldIndex = sideInfo.getSideTableInfo().getFieldList().indexOf(fieldName.trim());
 227                  object = SwitchUtil.getTarget(object, fields[fieldIndex]);
 228                  oneRow.put(fieldName.trim(), object);
 229              }
 230  
 231              String cacheKey = sideInfo.getEqualFieldList().stream()
 232 -                    .map(equalField -> oneRow.get(equalField))
 233 +                    .map(oneRow::get)
 234                      .map(Object::toString)
 235                      .collect(Collectors.joining("_"));
 236  
 237              tmpCache.computeIfAbsent(cacheKey, key -> Lists.newArrayList())
 238                      .add(oneRow);
 239          }
 240      }
