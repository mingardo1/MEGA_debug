  48   @Override public void handlerTaskRecord(TaskRecord record) {                                           
  49     RecordHelper helper = new RecordHelper(getWrapper(), record);                                        
  50     if (getWrapper().isSupportBP()) {                                                                    
  51       if (record.isBlock) {                                                                              
  52         helper.handleBlockRecord();                                                                      
  53       } else {                                                                                           
  54         helper.handleMultiRecord();                                                                      
  55       }                                                                                                  
  56     } else if (!getWrapper().isSupportBP()) {                                                            
  57       helper.handleNoSupportBPRecord();

  60     }

  64   public ThreadRecord createThreadRecord(TaskRecord record, int threadId, long startL, long endL) {      
  65     ThreadRecord tr;                                                                                     
  66     tr = new ThreadRecord();                                                                             
  67     tr.taskKey = record.filePath;                                                                        
  68     tr.threadId = threadId;                                                                              
  69     tr.startLocation = startL;                                                                           
  70     tr.isComplete = false;                                                                               
  71                                                                                                          
  72     tr.threadType = getEntity().getTaskType();                                                           
  73     //最后一个线程的结束位置即为文件的总长度                                                                                
  74     if (threadId == (record.threadNum - 1)) {                                                            
  75       endL = getEntity().getFileSize();                                                                  
  76     }                                                                                                    
  77     tr.endLocation = endL;                                                                               
  78     tr.blockLen = RecordUtil.getBlockLen(getEntity().getFileSize(), threadId, record.threadNum);         
  79     return tr;                                                                                           
  80   }

  82   @Override public TaskRecord createTaskRecord(int threadNum) {                                          
  83     TaskRecord record = new TaskRecord();                                                                
  84     record.fileName = getEntity().getFileName();                                                         
  85     record.filePath = getEntity().getFilePath();                                                         
  86     record.threadRecords = new ArrayList<>();                                                            
  87     record.threadNum = threadNum;                                                                        
  88                                                                                                          
  89     int requestType = getWrapper().getRequestType();                                                     
  90     if (requestType == ITaskWrapper.D_FTP || requestType == ITaskWrapper.D_FTP_DIR                       
  91         || requestType == ITaskWrapper.D_HTTP || requestType == ITaskWrapper.DG_HTTP) {                  
  92       record.isBlock = Configuration.getInstance().downloadCfg.isUseBlock();                             
  93     } else {                                                                                             
  94       record.isBlock = false;                                                                            
  95     }                                                                                                    
  96     record.taskType = getEntity().getTaskType();                                                         
  97     record.isGroupRecord = getEntity().isGroupChild();                                                   
  98     if (record.isGroupRecord) {                                                                          
  99       if (getEntity() instanceof DownloadEntity) {                                                       
 100         record.dGroupHash = ((DownloadEntity) getEntity()).getGroupHash();                               
 101       }                                                                                                  
 102     }                                                                                                    
 103                                                                                                          
 104     return record;                                                                                       
 105   }                                                                                                      
 106                                                                                                          
 107   @Override public int initTaskThreadNum() {                                                             
 108     int requestTpe = getWrapper().getRequestType();                                                      
 109     if (requestTpe == ITaskWrapper.U_HTTP                                                                
 110         || (requestTpe == ITaskWrapper.D_HTTP && (!getWrapper().isSupportBP())                           
 111         || ((HttpTaskOption) getWrapper().getTaskOption()).isChunked())) {                               
 112       return 1;                                                                                          
 113     }                                                                                                    
 114     int threadNum = Configuration.getInstance().downloadCfg.getThreadNum();                              
 115     return getEntity().getFileSize() <= IRecordHandler.SUB_LEN                                           
 116         || getEntity().isGroupChild()                                                                    
 117         || threadNum == 1                                                                                
 118         ? 1                                                                                              
 119         : threadNum;                                                                                     
 120   }
