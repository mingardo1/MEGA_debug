 282                      continue;

 320      private Operation toOperation(WriteMode writeMode) {
 321          switch (writeMode) {
 322              case INSERT:
 323                  return table.newInsert();
 324              case UPDATE:
 325                  return table.newUpdate();
 326              case UPSERT:
 327                  return table.newUpsert();
 328              default:
 329                  return table.newUpsert();
 330          }
