 250              while (oldVersion < newVersion) {                                                                     
 251                  switch (oldVersion) {                                                                             
 252                      case 1:                                                                                       
 253                          upgradeFrom1To2(db);                                                                      
 254                          oldVersion++;

 256                      case 2:                                                                                       
 257                          upgradeFrom2To3(db);                                                                      
 258                          oldVersion++;                                                                             
 259                          break;                                                                                    
 260                      case 3:                                                                                       
 261                          upgradeFrom3To4(db);                                                                      
 262                          oldVersion++;                                                                             
 263                          break;                                                                                    
 264                      case 4:                                                                                       
 265                          upgradeFrom4To5(db);                                                                      
 266                          oldVersion++;                                                                             
 267                          break;                                                                                    
 268                      case 5:                                                                                       
 269                          upgradeFrom5To6(db);                                                                      
 270                          oldVersion++;                                                                             
 271                          break;                                                                                    
 272                      default:                                                                                      
 273                          throw new IllegalStateException("not implemented");                                       
 274                  }                                                                                                 
 275              }                                                                                                     
 276                                                                                                                    
 277              mDatabaseCreated = true;                                                                              
 278              mDatabaseUpgraded = true;
