 297             private void updatePanelOffset(int progress) {                                               
 298                 int panelWidth = mSearchOverlay.getMeasuredWidth();                                      
 299                 int offset = (int) ((progress / 100f) * panelWidth);                                     
 300                 mSearchOverlay.setTranslationX(- panelWidth + offset);

 312                     });                                                                                  
 313                     oa.start();                                                                          
 314                     mOverlayPanelShowing = true;                                                         
 315                     mShowOverlayFeedback = false;                                                        
 316                 }                                                                                        
 317             }
