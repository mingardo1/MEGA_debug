  68          if(CollectionUtils.isEmpty(fieldList)){
  69              return "";
  70          }
  71          StringBuffer sb = new StringBuffer("");
  72          for(ReplaceInfo replaceInfo : fieldList){
  73  
  74              if(replaceInfo.getType() == EReplaceType.CONSTANT){
  75                  sb.append(replaceInfo.getParam());

  77              }
  78  
  79              if(replaceInfo.getType() == EReplaceType.FUNC){
  80                  sb.append(MD5Utils.getMD5String(buildStr(replaceInfo.getSubReplaceInfos(), refData)));
  81                  continue;
  82              }
  83              String replaceName = replaceInfo.getParam();
  84              if(!refData.containsKey(replaceName)){
  85                  throw new RuntimeException(String.format("build rowKey with field %s which value not found.", replüîµ
  86              }
  87              Object value = refData.get(replaceName);
  88              if(value instanceof Date){
  89                  SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMddHHmmss");
  90                  java.util.Date d = (java.util.Date) value;

 132      private List<ReplaceInfo> makeFormula(String formula){
 133          if(formula == null || formula.length() <= 0){
 134              return Lists.newArrayList();
 135          }
 136          List<ReplaceInfo> result = Lists.newArrayList();
 137          for(String meta: splitIgnoreQuotaBrackets(formula, "\\+")){
 138              Matcher matcher = Md5Operator.matcher(meta.trim());
 139              if(matcher.find()){
 140                  ReplaceInfo replaceInfo = new ReplaceInfo(EReplaceType.FUNC);
 141                  replaceInfo.setSubReplaceInfos(makeFormula(matcher.group(1)));
 142                  result.add(replaceInfo);
 143              } else {
 144                  result.add(getReplaceInfo(meta));
 145              }
 146          }
 147          return result;
 148      }
