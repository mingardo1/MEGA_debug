  98      public void setRowField(Row row, Integer index, BaseSideInfo sideInfo, String value) {
  99          Integer keyIndex = sideInfo.getSideFieldIndex().get(index);
 100          String classType = sideInfo.getSideTableInfo().getFieldClassList().get(keyIndex).getName();
 101          switch (classType){
 102              case "java.lang.Integer":
 103                  row.setField(index, Integer.valueOf(value));
 104                  break;
 105              case "java.lang.String":
 106                  row.setField(index, value);
 107                  break;
 108              case "java.lang.Double":
 109                  row.setField(index, Double.valueOf(value));
 110                  break;
 111              case "java.lang.Long":
 112                  row.setField(index, Long.valueOf(value));
 113                  break;
 114              case "java.lang.Byte":
 115                  row.setField(index, Byte.valueOf(value));
 116                  break;
 117              case "java.lang.Short":
 118                  row.setField(index, Short.valueOf(value));
 119                  break;
 120              case "java.lang.Float":
 121                  row.setField(index, Float.valueOf(value));
 122                  break;
 123              case "java.math.BigDecimal":
 124                  row.setField(index, BigDecimal.valueOf(Long.valueOf(value)));
 125                  break;
 126              case "java.sql.Timestamp":
 127                  row.setField(index, Timestamp.valueOf(value));
 128                  break;
 129              case "java.sql.Date":
 130                  row.setField(index, Date.valueOf(value));
 131                  break;
 132              default:
 133                  throw new RuntimeException("no support field type. the type: " + classType);
 134          }
