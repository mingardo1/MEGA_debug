 163      * Schedule a request which will be executed in the future. If you want to update an existing        
 164      * {@link JobRequest}, call {@link JobRequest#cancelAndEdit()}, update your parameters and call      
 165      * this method again. Calling this method with the same request multiple times without cancelling    
 166      * it is idempotent.                                                                                 
 167      *                                                                                                   
 168      * @param request The {@link JobRequest} which will run in the future.                               
 169      */                                                                                                  
 170     public void schedule(@NonNull JobRequest request) {                                                  
 171         if (mJobCreatorHolder.isEmpty()) {                                                               
 172             CAT.w("you haven't registered a JobCreator with addJobCreator(), it's likely that your job neðŸ”µ
 173         }                                                                                                
 174                                                                                                          
 175         if (request.getScheduledAt() > 0) {                                                              
 176             return;                                                                                      
 177         }                                                                                                
 178                                                                                                          
 179         if (request.isUpdateCurrent()) {                                                                 
 180             cancelAllForTag(request.getTag());                                                           
 181         }                                                                                                
 182                                                                                                          
 183         JobProxy.Common.cleanUpOrphanedJob(mContext, request.getJobId());                                
 184                                                                                                          
 185         JobApi jobApi = request.getJobApi();                                                             
 186         boolean periodic = request.isPeriodic();                                                         
 187         boolean flexSupport = periodic && jobApi.isFlexSupport() && request.getFlexMs() < request.getInteðŸ”µ
 188                                                                                                          
 189         request.setScheduledAt(System.currentTimeMillis());                                              
 190         request.setFlexSupport(flexSupport);                                                             
 191         mJobStorage.put(request);                                                                        
 192                                                                                                          
 193         try {                                                                                            
 194             scheduleWithApi(request, jobApi, periodic, flexSupport);                                     
 195             return;                                                                                      
 196         } catch (JobProxyIllegalStateException e) {                                                      
 197             // try again below, the other cases stop                                                     
 198                                                                                                          
 199         } catch (Exception e) {                                                                          
 200             // if something fails, don't keep the job in the database, it would be rescheduled later     
 201             mJobStorage.remove(request);                                                                 
 202             throw e;                                                                                     
 203         }                                                                                                
 204                                                                                                          
 205         try {                                                                                            
 206             // try to reload the proxy                                                                   
 207             jobApi.invalidateCachedProxy();                                                              
 208                                                                                                          
 209             scheduleWithApi(request, jobApi, periodic, flexSupport);                                     
 210             return;                                                                                      
 211         } catch (Exception e) {                                                                          
 212             if (jobApi == JobApi.V_14 || jobApi == JobApi.V_19) {                                        
 213                 // at this stage we cannot do anything                                                   
 214                 mJobStorage.remove(request);                                                             
 215                 throw e;                                                                                 
 216             } else {                                                                                     
 217                 jobApi = JobApi.V_19.isSupported(mContext) ? JobApi.V_19 : JobApi.V_14; // try one last tðŸ”µ
 218             }                                                                                            
 219         }                                                                                                
 220                                                                                                          
 221         try {                                                                                            
 222             scheduleWithApi(request, jobApi, periodic, flexSupport);                                     
 223         } catch (Exception e) {                                                                          
 224             // if something fails, don't keep the job in the database, it would be rescheduled later     
 225             mJobStorage.remove(request);                                                                 
 226             throw e;                                                                                     
 227         }                                                                                                
 228     }

 433          if (job != null && !job.isFinished() && !job.isCanceled()) {                                              
 434              CAT.i("Cancel running %s", job);                                                                      
 435              job.cancel(true);                                                                                     
 436              return true;                                                                                          
 437          } else {                                                                                                  
 438              return false;                                                                                         
 439          }

 442      private synchronized int cancelAllInner(@Nullable String tag) {                                               
 443          int canceled = 0;                                                                                         
 444                                                                                                                    
 445          Set<JobRequest> requests = mJobStorage.getAllJobRequests(tag, true);                                      

 446          for (JobRequest request : requests) {                                                                     
 447              if (cancelInner(request)) {                                                                           
 448                  canceled++;                                                                                       
 449              }                                                                                                     
 450          }                                                                                                         
 451                                                                                                                    
 452          //noinspection ConstantConditions                                                                         
 453          Set<Job> jobs = TextUtils.isEmpty(tag) ? getAllJobs() : getAllJobsForTag(tag);                            
 454          for (Job job : jobs) {                                                                                    
 455              if (cancelInner(job)) {                                                                               
 456                  canceled++;                                                                                       
 457              }                                                                                                     
 458          }                                                                                                         
 459          return canceled;                                                                                          
 460      }
