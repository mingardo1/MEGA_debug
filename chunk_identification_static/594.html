 297     protected void buildSqlTypes(List<Class> fieldTypeArray) {
 298 
 299         int[] tmpFieldsType = new int[fieldTypeArray.size()];
 300         for (int i = 0; i < fieldTypeArray.size(); i++) {
 301             String fieldType = fieldTypeArray.get(i).getName();
 302             if (fieldType.equals(Integer.class.getName())) {
 303                 tmpFieldsType[i] = Types.INTEGER;
 304             }else if (fieldType.equals(Boolean.class.getName())) {
 305                 tmpFieldsType[i] = Types.BOOLEAN;
 306             }else if (fieldType.equals(Long.class.getName())) {
 307                 tmpFieldsType[i] = Types.BIGINT;
 308             } else if (fieldType.equals(Byte.class.getName())) {
 309                 tmpFieldsType[i] = Types.TINYINT;
 310             } else if (fieldType.equals(Short.class.getName())) {
 311                 tmpFieldsType[i] = Types.SMALLINT;
 312             } else if (fieldType.equals(String.class.getName())) {
 313                 tmpFieldsType[i] = Types.CHAR;
 314             } else if (fieldType.equals(Byte.class.getName())) {
 315                 tmpFieldsType[i] = Types.BINARY;
 316             } else if (fieldType.equals(Float.class.getName())) {
 317                 tmpFieldsType[i] = Types.FLOAT;
 318             } else if (fieldType.equals(Double.class.getName())) {
 319                 tmpFieldsType[i] = Types.DOUBLE;
 320             } else if (fieldType.equals(Timestamp.class.getName())) {
 321                 tmpFieldsType[i] = Types.TIMESTAMP;
 322             } else if (fieldType.equals(BigDecimal.class.getName())) {
 323                 tmpFieldsType[i] = Types.DECIMAL;
 324             } else if (fieldType.equals(Date.class.getName())) {
 325                 tmpFieldsType[i] = Types.DATE;
 326             } else {
 327                 throw new RuntimeException("no support field type for sql. the input type:" + fieldType);
 328             }
 329         }
 330 
 331         this.sqlTypes = tmpFieldsType;
 332     }

 165      protected void buildSqlTypes(List<Class> fieldTypeArray) {
 166  
 167          int[] tmpFieldsType = new int[fieldTypeArray.size()];
 168          for (int i = 0; i < fieldTypeArray.size(); i++) {
 169              String fieldType = fieldTypeArray.get(i).getName();
 170              if (fieldType.equals(Integer.class.getName())) {
 171                  tmpFieldsType[i] = Types.INTEGER;
 172              }else if (fieldType.equals(Boolean.class.getName())) {
 173                  tmpFieldsType[i] = Types.BOOLEAN;
 174              }else if (fieldType.equals(Long.class.getName())) {
 175                  tmpFieldsType[i] = Types.BIGINT;
 176              } else if (fieldType.equals(Byte.class.getName())) {
 177                  tmpFieldsType[i] = Types.TINYINT;
 178              } else if (fieldType.equals(Short.class.getName())) {
 179                  tmpFieldsType[i] = Types.SMALLINT;
 180              } else if (fieldType.equals(String.class.getName())) {
 181                  tmpFieldsType[i] = Types.CHAR;
 182              } else if (fieldType.equals(Byte.class.getName())) {
 183                  tmpFieldsType[i] = Types.BINARY;
 184              } else if (fieldType.equals(Float.class.getName())) {
 185                  tmpFieldsType[i] = Types.FLOAT;
 186              } else if (fieldType.equals(Double.class.getName())) {
 187                  tmpFieldsType[i] = Types.DOUBLE;
 188              } else if (fieldType.equals(Timestamp.class.getName())) {
 189                  tmpFieldsType[i] = Types.TIMESTAMP;
 190              } else if (fieldType.equals(BigDecimal.class.getName())) {
 191                  tmpFieldsType[i] = Types.DECIMAL;
 192              } else if (fieldType.equals(Date.class.getName())) {
 193                  tmpFieldsType[i] = Types.DATE;
 194              } else {
 195                  throw new RuntimeException("no support field type for sql. the input type:" + fieldType);
 196              }
 197          }
 198  
 199          this.sqlTypes = tmpFieldsType;
 200      }
