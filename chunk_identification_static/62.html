 176               if (TextUtils.isEmpty(value)){

 181               field.setLong(this, Long.parseLong(value));                                                
 182             } else if (type == boolean.class || type == Boolean.class) {                                 
 183               field.setBoolean(this, Boolean.parseBoolean(value));

 153       boolean isDownload = this instanceof DownloadConfig;                                               
 154       File file = new File(                                                                              
 155           AriaManager.APP.getFilesDir().getPath() + (isDownload ? DOWNLOAD_CONFIG_FILE                   
 156               : UPLOAD_CONFIG_FILE));                                                                    
 157       if (file.exists()) {                                                                               
 158         Properties properties = CommonUtil.loadConfig(file);                                             
 159         List<Field> fields = CommonUtil.getAllFields(getClass());                                        
 160         try {                                                                                            
 161           for (Field field : fields) {                                                                   
 162             int m = field.getModifiers();                                                                
 163             if (Modifier.isFinal(m) || Modifier.isStatic(m)) {                                           
 164               continue;                                                                                  
 165             }                                                                                            
 166             field.setAccessible(true);                                                                   
 167             String value = properties.getProperty(field.getName());                                      
 168             Class<?> type = field.getType();                                                             
 169             if (type == String.class) {                                                                  
 170               field.set(this, value);                                                                    
 171             } else if (type == int.class || type == Integer.class) {                                     
 172               field.setInt(this, Integer.parseInt(value));                                               
 173             } else if (type == float.class || type == Float.class) {                                     
 174               field.setFloat(this, Float.parseFloat(value));                                             
 175             } else if (type == double.class || type == Double.class) {                                   
 176               if (TextUtils.isEmpty(value)){                                                             
 177                 value = "0.0";                                                                           
 178               }                                                                                          
 179               field.setDouble(this, Double.parseDouble(value));                                          
 180             } else if (type == long.class || type == Long.class) {                                       
 181               field.setLong(this, Long.parseLong(value));                                                
 182             } else if (type == boolean.class || type == Boolean.class) {                                 
 183               field.setBoolean(this, Boolean.parseBoolean(value));                                       
 184             }                                                                                            
 185           }                                                                                              
 186         } catch (IllegalAccessException e) {                                                             
 187           e.printStackTrace();                                                                           
 188         }                                                                                                
 189       }
